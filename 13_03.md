```
n = int(input()) #Вводится любая переменная.
r = bin(n) [2:] #Выводит переменную n в двоичную систему.
print(r) #Вывод числа.
if n % 2 == 0: #Если число четное то выведи 01
    r += '01'
else: #Если число не четное выведи 10
    r += '10'
print(r)
```
